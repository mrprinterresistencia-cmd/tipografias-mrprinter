      <h1>Buscador de nombre — Vista previa en tus tipografías</h1>

        <input id="nameInput" type="text" placeholder="Escribí el nombre… (ej: Sofía)" autocomplete="off" />

        Escribí el nombre una vez y se muestra automáticamente en 4 tipografías (OTF).

      </div>

    </div>

  <script>

    // =========================

    // 3) CONFIGURACIÓN DE FUENTES

    //    family debe coincidir con font-family de @font-face

    // =========================

    const FONTS = [

      { label: "Cream Cake Bold", family: "Cream Cake Bold", size: 54 },

      { label: "Muthiara Regular", family: "Muthiara Regular", size: 54 },

      { label: "Fuente 3", family: "Fuente 3", size: 48 },

      { label: "Fuente 4", family: "Fuente 4", size: 48 },

    ];


    const input = document.getElementById("nameInput");

    const previews = document.getElementById("previews");

    const clearBtn = document.getElementById("clearBtn");

    const copyBtn = document.getElementById("copyBtn");


    function escapeHtml(str) {

      return str.replace(/[&<>"']/g, m => ({

        "&":"&amp;",

        "<":"&lt;",

        ">":"&gt;",

        '"':"&quot;",

        "'":"&#039;"

      }[m]));

    }


    function render(text) {

      const value = (text || "").trim() || "Sofía";

      previews.innerHTML = "";


      for (const f of FONTS) {

        const div = document.createElement("div");

        div.className = "preview";

        div.innerHTML = `

          <div class="meta">

            <div class="fontname">${escapeHtml(f.label)}</div>

            <div class="size">${f.size}px</div>

          </div>

          <div class="sample" style="font-family:'${escapeHtml(f.family)}'; font-size:${f.size}px;">

            ${escapeHtml(value)}

          </div>

        `;

        previews.appendChild(div);

      }

    }


    input.addEventListener("input", e => render(e.target.value));


    clearBtn.addEventListener("click", () => {

      input.value = "";

      render("");

      input.focus();

    });


    copyBtn.addEventListener("click", async () => {

      const value = (input.value || "").trim() || "Sofía";

      try { await navigator.clipboard.writeText(value); } catch (e) {}

    });


    render("");

  </script>
